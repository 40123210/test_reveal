from browser import console

# 準備繪圖畫布
canvas = doc["plotarea"]
ctx = canvas.getContext("2d")

# 開始畫直線
for i in range(11):
    ctx.beginPath()
    ctx.lineWidth = 3
    ctx.moveTo(0,0+i*25 )
    ctx.lineTo(200,0+i*25 )
    ctx.strokeStyle = "#7700FF"
    ctx.stroke()
for i in range(11):
    ctx.beginPath()
    ctx.lineWidth = 5
    ctx.moveTo(0+i*20, 0)
    ctx.lineTo(0+i*20, 250)
    ctx.strokeStyle = "#FF0066"
    ctx.stroke()


</script>
